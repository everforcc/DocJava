~~~

# crud
s.add( x )

s.remove( x )
# 元素不存在，不会发生错误。
# pop 方法会对集合进行无序的排列，然后将这个无序排列集合的左面第一个元素进行删除。
s.discard( x )

# 随机删除一个元素
s.pop() 

# 删除集合中所有元素
s.clear()
