### 总订单金额

~~~sql
select sum(t.total_price) as amount
from znkj_purchase_order t
where t.del_flag = 0
  and t.`status` in (3, 4, 5);
~~~

### 售卖系列订单金额

~~~sql
select sum(t.total_price) as amount
from znkj_purchase_order t
where t.del_flag = 0
  and t.`status` in (3, 4, 5)
  and t.product_type = 1;

~~~

### 租赁系列订单金额

~~~sql
select sum(t.total_price) as amount
from znkj_purchase_order t
where t.del_flag = 0
  and t.`status` in (3, 4, 5)
  and t.product_type = 2;
~~~

### 保证金

~~~sql
SELECT sum(t.earnest_money) FROM znkj_dealer t WHERE t.del_flag=0;
~~~

### 设置充值总额

~~~sql
select sum(c.amount) from znkj_coin c
        where
            c.trading_tag = 1
            and c.trading_type = 2
            and c.del_flag = 0;
~~~

### 设置消费总额

~~~sql
select abs(sum(amount)) from znkj_coin c
        where
        c.trading_tag = 4
        and c.trading_type = 1
        and c.del_flag = 0;
~~~

### 设置帐户总余额

~~~sql
SELECT sum(t.balance) FROM znkj_dealer t WHERE t.del_flag=0;
~~~
